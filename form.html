    <script>

        const canvas = document.getElementById('canvas-bg');

        const ctx = canvas.getContext('2d');

        let particlesArray;

        

        // 鼠标交互对象

        let mouse = {

            x: null,

            y: null,

            radius: 150 // 鼠标吸附半径

        }


        window.addEventListener('mousemove', function(event) {

            mouse.x = event.x;

            mouse.y = event.y;

        });


        function resizeCanvas() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

            init();

        }

        window.addEventListener('resize', resizeCanvas);


        class Particle {

            constructor() {

                this.x = Math.random() * canvas.width;

                this.y = Math.random() * canvas.height;

                this.size = Math.random() * 3 + 1; // [增强] 粒子变大

                this.speedX = (Math.random() * 1) - 0.5; // [增强] 速度稍快

                this.speedY = (Math.random() * 1) - 0.5;

            }

            update() {

                this.x += this.speedX;

                this.y += this.speedY;


                // 边界反弹

                if (this.x > canvas.width || this.x < 0) this.speedX = -this.speedX;

                if (this.y > canvas.height || this.y < 0) this.speedY = -this.speedY;


                // 鼠标互动

                let dx = mouse.x - this.x;

                let dy = mouse.y - this.y;

                let distance = Math.sqrt(dx*dx + dy*dy);

                if (distance < mouse.radius) {

                    if (mouse.x < this.x && this.x < canvas.width - this.size * 10) {

                        this.x += 2;

                    }

                    if (mouse.x > this.x && this.x > this.size * 10) {

                        this.x -= 2;

                    }

                    if (mouse.y < this.y && this.y < canvas.height - this.size * 10) {

                        this.y += 2;

                    }

                    if (mouse.y > this.y && this.y > this.size * 10) {

                        this.y -= 2;

                    }

                }

            }

            draw() {

                ctx.fillStyle = 'rgba(0, 243, 255, 1)'; // [增强] 纯亮青色，不透明

                ctx.beginPath();

                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);

                ctx.fill();

            }

        }


        function init() {

            particlesArray = [];

            let numberOfParticles = (canvas.width * canvas.height) / 9000; // [增强] 粒子密度增加

            for (let i = 0; i < numberOfParticles; i++) {

                particlesArray.push(new Particle());

            }

        }


        function animate() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            for (let i = 0; i < particlesArray.length; i++) {

                particlesArray[i].update();

                particlesArray[i].draw();

                

                // 连线逻辑

                for (let j = i; j < particlesArray.length; j++) {

                    const dx = particlesArray[i].x - particlesArray[j].x;

                    const dy = particlesArray[i].y - particlesArray[j].y;

                    const distance = Math.sqrt(dx * dx + dy * dy);

                    

                    // [增强] 连线距离更长，线条更明显

                    if (distance < 120) {

                        ctx.beginPath();

                        // 线条透明度更高 (0.5 base)

                        ctx.strokeStyle = `rgba(0, 243, 255, ${0.5 - distance/300})`;

                        ctx.lineWidth = 1.5; // [增强] 线条变粗

                        ctx.moveTo(particlesArray[i].x, particlesArray[i].y);

                        ctx.lineTo(particlesArray[j].x, particlesArray[j].y);

                        ctx.stroke();

                    }

                }

            }

            requestAnimationFrame(animate);

        }


        resizeCanvas();

        animate();

    </script>
